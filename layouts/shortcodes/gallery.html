{{- /* Dependency-free photo gallery: masonry grid + native lightbox. {{< gallery >}} reads content bundle resources at gallery/* {{< gallery match="photos/*" >}} custom glob (page bundle, else /assets) Captions are opt-in via the page's `resources` front matter: resources: - src: "gallery/sunset.jpg" params: { caption: "Golden hour on Georgian Bay" } */ -}} {{- $match := .Get "match" | default "gallery/*" -}} {{- $found := .Page.Resources.Match $match -}} {{- if not $found }}{{ $found = resources.Match $match }}{{ end -}} {{- /* Keep only formats Hugo can process; skips HEIC and other non-image files. */ -}} {{- $ok := slice "jpeg" "jpg" "png" "gif" "tiff" "bmp" "webp" -}} {{- $imgs := slice -}} {{- range $found -}} {{- if in $ok (lower .MediaType.SubType) -}}{{ $imgs = $imgs | append . }}{{- end -}} {{- end -}} {{- with $imgs -}}
{{- range $i, $img := . -}} {{- $thumb := $img.Resize "800x q85" -}} {{- $full := cond (gt $img.Width 1920) ($img.Resize "1920x q85") $img -}} {{- $cap := $img.Params.caption -}} {{- end -}}
